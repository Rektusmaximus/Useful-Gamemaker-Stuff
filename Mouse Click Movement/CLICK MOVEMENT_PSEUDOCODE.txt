OBJECT: PLAYER
EVENT: CREATE
	DEFINE newPositionX TO round(x)

OBJECT: PLAYER
EVENT: STEP
	IF (x IS NOT EQUAL TO newPositionX)
		SET IMAGE SPEED to DESIRED SPEED
		IF (x IS LESS THAN newPositionX)
			SET X += 1
			SET IMAGE SPEED TO DESIRED SPEED
			SET IMAGE XSCALE TO 1
		IF (X IS GREATER THAN newPositionX)
			SET X -=1
			SET IMAGE SPEED TO DESIRED SPEED
			SET IMAGE XSCALE TO -1
	ELSE
		SET IMAGE SPEED TO 0;

OBJECT: PLAYER
EVENT: GLOBAL LEFT PRESSED
	SET newPositionX TO round(mouse_x)
